# configuration file /etc/nginx/conf.d/yui-quant-lab-zones.conf:
# 放到 /etc/nginx/conf.d/（Ubuntu 預設會在 http{} 內 include conf.d/*.conf）。
#
# 為什麼要獨立這一檔？
# - Nginx 規定 limit_req_zone 只能寫在 http 層，不能寫在 sites-available 的 server{} 片段裡。
# - 這裡只放「共用漏桶」定義；實際套用在哪幾條 URL，請看 deploy/nginx-yui-quant-lab.conf。
#
# 設計理念（白話）：
# - 垃圾流量、掃描、誤打，應盡量擋在 Nginx，別讓 Gunicorn（單 worker）吃爆。
# - 這不是高併發配額，而是「同一個來源 IP」在短時間內別太誇張」的保護網。
#
# 調整方式：
# - rate=30r/m 表示平均每分鐘約 30 次（平滑後）；burst 在 site 檔裡設定。
# - 若你的 TradingView / 券商回呼頻率真的更高，再微調數字即可。

limit_req_zone $binary_remote_addr zone=yui_webhooks:10m rate=30r/m;

# configuration file /etc/nginx/sites-enabled/yui-quant-lab:
# Site fragment: copy to /etc/nginx/sites-available/yui-quant-lab (or include from
# conf.d). Enable with symlink under sites-enabled, then: nginx -t && systemctl reload nginx
#
# Flow: public :80 -> proxy_pass 127.0.0.1:8000 (Gunicorn). TLS/domain: add server block
# later (e.g. certbot) on 443.
#
# --- Rate limit（正式環境建議一定要開）---
# 1) 先把 deploy/nginx-yui-quant-lab-zones.conf 複製到 /etc/nginx/conf.d/（檔名自訂即可）。
# 2) 再啟用本 site。否則 nginx -t 會報「unknown limit_req_zone」。
#
# 為什麼 webhook 路徑要限速？
# - 對外 URL 一定會被掃描；我們又是單 worker（Gunicorn workers=1），後端經不起硬撞。
#
# 為什麼這裡「不要」用 limit_req 的 nodelay？
# - nodelay 會讓 burst 內的大量請求「立刻」全部轉進後端；在 workers=1 時，
#   容易變成瞬間排隊/CPU 尖峰（JSON 解析、寫檔、決策都在同一個 process）。
# - 拿掉 nodelay：Nginx 會用漏桶把尖峰「抹平」一點，對單 worker 比較溫和。
# - TradingView 偶爾重送幾次，burst 仍留著緩衝，不會太嚴。

server {
    listen 80;
    listen [::]:80;
    server_name _;

    client_max_body_size 256k;

    location ~ ^/(webhook|tv-webhook|fill-result|order-event|telegram/webhook)$ {
        limit_req zone=yui_webhooks burst=10;
        proxy_pass http://127.0.0.1:8000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 120s;
    }

    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 120s;
    }
}
