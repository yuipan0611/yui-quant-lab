from flask import Flask, request, jsonify

app = Flask(__name__)

# 這裡定義 TradingView alert 必須帶的欄位
REQUIRED_FIELDS = [
    "symbol",
    "signal",
    "price",
    "breakout_level",
    "delta_strength",
]


@app.route("/health", methods=["GET"])
def health():
    # 健康檢查用，讓你可以快速確認服務是否活著
    return jsonify({"status": "ok"}), 200


@app.route("/webhook", methods=["POST"])
def webhook():
    """
    接收 TradingView webhook 的主入口
    流程：
    1) 先安全解析 JSON
    2) 檢查必要欄位
    3) 成功就印出資料並回傳 success
    """

    # 安全 JSON parsing：
    # - silent=True：如果 body 不是合法 JSON，不會直接丟 exception
    # - 會回傳 None，讓我們自己回應清楚的錯誤訊息
    payload = request.get_json(silent=True)

    if payload is None:
        return (
            jsonify(
                {
                    "status": "error",
                    "message": "Invalid or missing JSON body. Please send Content-Type: application/json with valid JSON.",
                }
            ),
            400,
        )

    # 再確認 JSON 根結構是物件（dict），不是 list/string
    if not isinstance(payload, dict):
        return (
            jsonify(
                {
                    "status": "error",
                    "message": "JSON payload must be an object (key-value pairs).",
                }
            ),
            400,
        )

    # 檢查哪些必要欄位缺少
    missing_fields = [field for field in REQUIRED_FIELDS if field not in payload]

    if missing_fields:
        return (
            jsonify(
                {
                    "status": "error",
                    "message": "Missing required fields.",
                    "missing_fields": missing_fields,
                    "required_fields": REQUIRED_FIELDS,
                }
            ),
            400,
        )

    # 到這裡表示資料完整，取出欄位（也方便後續加商業邏輯）
    symbol = payload["symbol"]
    signal = payload["signal"]
    price = payload["price"]
    breakout_level = payload["breakout_level"]
    delta_strength = payload["delta_strength"]

    # 先用 print 做最小可用版的日誌輸出
    print("=== TradingView Alert Received ===")
    print(f"symbol: {symbol}")
    print(f"signal: {signal}")
    print(f"price: {price}")
    print(f"breakout_level: {breakout_level}")
    print(f"delta_strength: {delta_strength}")
    print(f"raw_payload: {payload}")

    return (
        jsonify(
            {
                "status": "success",
                "message": "Webhook received successfully.",
                "data": {
                    "symbol": symbol,
                    "signal": signal,
                    "price": price,
                    "breakout_level": breakout_level,
                    "delta_strength": delta_strength,
                },
            }
        ),
        200,
    )


if __name__ == "__main__":
    # debug=True 方便本機開發看錯誤；上線建議關掉
    app.run(host="0.0.0.0", port=5000, debug=True)
